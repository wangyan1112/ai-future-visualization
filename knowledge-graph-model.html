knowledge_graph.py - 知识图谱核心类 
from neo4j import GraphDatabase
from langchain.embeddings import OpenAIEmbeddings
import json

class QAKnowledgeGraph:
    def __init__(self, uri, user, password):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))
        self.embeddings = OpenAIEmbeddings()
    
    # 导入API文档
    def import_openapi(self, spec_path):
        with open(spec_path) as f:
            spec = json.load(f)
        for path, methods in spec['paths'].items():
            for method, details in methods.items():
                self._create_api_node(path, method, details)
    
    # 影响分析 - 核心功能
    def impact_analysis(self, api_path):
        query = """
        MATCH (api:API {path: $path})
        OPTIONAL MATCH (api)-[:DEPENDS_ON*1..3]->(dep:API)
        OPTIONAL MATCH (api)<-[:TESTS]-(test:TestCase)
        OPTIONAL MATCH (api)<-[:AFFECTS]-(bug:Bug)
        RETURN api, collect(DISTINCT dep) as dependencies,
               collect(DISTINCT test) as tests,
               collect(DISTINCT bug) as bugs
        """
        return self._execute(query, {'path': api_path})
    
    # 智能测试推荐
    def recommend_tests(self, changed_files):
        affected_apis = self._find_affected_apis(changed_files)
        query = """
        MATCH (api:API)-[:TESTS]->(test:TestCase)
        WHERE api.path IN $apis
        RETURN test ORDER BY test.priority
        """
        return self._execute(query, {'apis': affected_apis})
cli.py - 命令行工具 
import click
from knowledge_graph import QAKnowledgeGraph

@click.group()
def cli():
    """QA知识图谱命令行工具"""
    pass

@cli.command()
@click.argument('api_path')
def impact(api_path):
    """分析API变更影响范围"""
    kg = QAKnowledgeGraph.from_env()
    result = kg.impact_analysis(api_path)
    click.echo(f"🔍 影响分析: {api_path}")
    click.echo(f"  📦 依赖API: {len(result['dependencies'])}")
    click.echo(f"  🧪 相关测试: {len(result['tests'])}")
    click.echo(f"  🐛 历史缺陷: {len(result['bugs'])}")

@cli.command()
def suggest():
    """基于Git变更推荐测试用例"""
    changed = get_git_changed_files()
    kg = QAKnowledgeGraph.from_env()
    tests = kg.recommend_tests(changed)
    click.echo("📋 推荐执行的测试用例:")
    for t in tests:
        click.echo(f"  [{t['priority']}] {t['name']}")
